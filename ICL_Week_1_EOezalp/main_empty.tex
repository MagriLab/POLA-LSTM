	\frametitle{Lorenz System}
	\begin{align*}
	    \frac{dx}{dt} &= \sigma(y-x)\\
	    \frac{dy}{dt} &= x(\rho-z) -y \\
	    \frac{dz}{dt} &= xy -\beta z 
	\end{align*}
	\begin{itemize}
	       \item$ \sigma = 10, \beta = 8/3, \rho = 28 $
	    \item Initial condition $(0, 1, 1)$
	    \item Time: $[0, 100], \Delta t = 0.001$
	\end{itemize}

\end{frame}
